import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# Data Generation and Analysis
np.random.seed(42)
departments_list = ['Marketing', 'HR', 'Sales', 'R&D', 'Operations', 'Finance']
data = {
    'employee_id': [f'EMP{i:03d}' for i in range(1, 101)],
    'department': np.random.choice(departments_list, 100),
    'region': np.random.choice(['Europe', 'Latin America', 'Middle East'], 100),
    'performance_score': np.random.uniform(60, 99, 100)
}
df = pd.DataFrame(data)
operations_count = df[df['department'] == 'Operations'].shape[0]
print(f"Frequency count for 'Operations': {operations_count}")
